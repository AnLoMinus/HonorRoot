[שִׁיר 4 — עֵץ עִם שֵׁמוֹת]

[Intro]
גִּיטָרָה נוֹגַעַת בְּשֶׁקֶט, חָלִיל מְסַפֵּר,
דַּפִּים שֶׁל אַלְבּוֹם נִפְתָּחִים כְּמוֹ שַׁעַר אַחֵר.
עַל כָּל תְּמוּנָה יְשָׁנָה יֵשׁ אוֹר שֶׁלֹּא נִגְמַר,
כָּל פָּנִים שֶׁנִּשְׁתַּקּוּ עוֹד שָׁרִים לִי מִן הֶעָבָר.

[בַּיִת 1]
אֲנִי פּוֹתֵחַ אַלְבּוֹם, וְהַלֵּב נִפְתָּח אִתּוֹ,
שֵׁם לְיַד שֵׁם, כְּמוֹ נֵר שֶׁמַּדְלִיק נֵר בְּתוֹכוֹ.
סָבָא בְּחִיּוּךְ שֶׁל מֶלַח, סָבְתָא בְּמַבָּט שֶׁל תְּפִלָּה,
אַהֲבָה שֶׁלֹּא עָשְׂתָה רַעַשׁ, אַךְ הֶחֱזִיקָה מִשְׁפָּחָה גְּדוֹלָה.
כָּל קְמָט בַּתְּמוּנָה הוּא מַפָּה שֶׁל מִלְחָמָה,
כָּל עַיִן שֶׁהִבִּיטָה רָחוֹק הִשְׁאִירָה בִּי נְשָׁמָה.
זֶה פַּאנְץ׳ רִאשׁוֹן: שֵׁם מִשְׁפָּחָה אֵינֶנּוּ רַק סִימָן,
זֶה דּוֹר שֶׁלָּחַ לְךָ מִכְתָּב עַל גַּב הַזְּמַן.
זֶה פַּאנְץ׳ שֵׁנִי: מִי שֶׁחוֹתֵךְ אֶת הַשֹּׁרֶשׁ בְּיָד,
מְגַלֶּה שֶׁהֶעָנָף הָכִי גָּבוֹהַּ נִשְׁאָר לְבַד.

[פִּזְמוֹן]
כָּל שֵׁם בֶּעָנָף הוּא נֵר בְּתוֹךְ הַדָּם,
כָּל דּוֹר שֶׁהָיָה כָּאן עוֹד לוֹחֵשׁ לִי: קוּם גַּם.
אֲנִי לֹא רַק עַצְמִי, אֲנִי דֶּרֶךְ שֶׁעָבְרָה,
עֵץ עִם שֵׁמוֹת בַּלֵּב — וְשֹׁרֶשׁ שֶׁל בְּרָכָה.

[בַּיִת 2]
אַבָּא הוּא עָנָף שֶׁנִּשָּׂא עַל רוּחַ וְעָמָל,
אִמָּא הִיא עָלֶה שֶׁמַּצְלִיל לִי חֹם בְּקוֹל קָטָן.
וַאֲנִי בֵּין הַשְּׁנַיִם, כְּמוֹ פְּרִי שֶׁלּוֹמֵד לְהַבְשִׁיל,
לִקַּחְתִּי מֵהֶם אוֹר וּמְנַסֶּה לְהַחְזִיר לַשְּׁבִיל.
יְלָדִים עוֹד יָבוֹאוּ, יִשְׁאֲלוּ מֵאֵיפֹה הַשִּׁיר,
וַאֲנִי אֶפְתַּח לָהֶם עֵץ וְאֹמַר: זֶה הַנִּגּוּן הֶעָשִׁיר.
זֶה פַּאנְץ׳ שְׁלִישִׁי: יְרוּשָּׁה אֵינָהּ רַק כֶּסֶף וְקִיר,
יֵשׁ יְרוּשָּׁה שֶׁהִיא דֶּרֶךְ, וְהִיא עוֹשָׂה עָנִי לְעָשִׁיר.
זֶה פַּאנְץ׳ רְבִיעִי: כְּשֶׁאַתָּה מְכַבֵּד אֶת מִי שֶׁקָּדַם,
אַתָּה מַדְלִיק מָחָר בְּתוֹךְ הַיּוֹם שֶׁלֹּא נִרְדַּם.

[פִּזְמוֹן]
כָּל שֵׁם בֶּעָנָף הוּא נֵר בְּתוֹךְ הַדָּם,
כָּל דּוֹר שֶׁהָיָה כָּאן עוֹד לוֹחֵשׁ לִי: קוּם גַּם.
אֲנִי לֹא רַק עַצְמִי, אֲנִי דֶּרֶךְ שֶׁעָבְרָה,
עֵץ עִם שֵׁמוֹת בַּלֵּב — וְשֹׁרֶשׁ שֶׁל בְּרָכָה.

[בַּיִת 3]
שֵׁם הוּא לֹא אוֹתִיּוֹת בִּלְבַד, הוּא מַסָּע בִּתְפִירָה,
הוּא דֶּמַע שֶׁנֶּהְפַּךְ לְנִגּוּן, הוּא מַפָּה שֶׁל גְּבוּרָה.
הוּא בַּקּוֹ, הוּא יְרוּשָׁלַיִם, הוּא שֻׁלְחָן, הוּא שַׁבָּת,
הוּא קִדּוּשׁ בְּקוֹל רוֹעֵד, הוּא עִנְיָן שֶׁל בְּרִית וְדַעַת.
הוּא סִפּוּר שֶׁל מַעֲבָרִים, שֶׁל מִזְוָדוֹת וְתִקְוָה,
שֶׁל אֵם שֶׁמְּבָרֶכֶת יֶלֶד וְאָב שֶׁמַּחְזִיק עֲנָוָה.
זֶה פַּאנְץ׳ חֲמִישִׁי: עָבָר שֶׁמְּכַבְּדִים לֹא נִקְבָּר,
הוּא הוֹפֵךְ לְאוֹר עָתִיד וְעוֹמֵד לְךָ כְּמִשְׁמָר.
זֶה פַּאנְץ׳ שִׁשִּׁי: הַדּוֹר הַבָּא לֹא שׁוֹאֵל מַה דִּבַּרְתָּ,
הוּא מַבִּיט אֵיךְ כִּבַּדְתָּ אֶת הַשֵּׁם שֶׁקִּבַּלְתָּ.

[גֶּשֶׁר]
אֲנִי לֹא רַק אֲנִי — אֲנִי הֶמְשֵׁךְ שֶׁל אוֹר,
דּוֹר מְדַבֵּר עִם דּוֹר, וְהַדָּם לוֹמֵד לִזְכֹּר.
אֲנִי לֹא רַק עַכְשָׁו — אֲנִי גֶּשֶׁר בֵּין זְמַנִּים,
מִי שֶׁשּׁוֹמֵר עַל שֵׁם — שׁוֹמֵר עַל בֵּית חַיִּים.
אִם לִפְעָמִים שָׁכַחְתִּי לְהַבִּיט לְאָחוֹר,
הַיּוֹם אֲנִי חוֹזֵר לָעֵץ וּמְחַבֵּק אֶת הָאוֹר.
לֹא בִּשְׁבִיל לְהִתְקַטֵּן, אֶלָּא כְּדֵי לְהִתְחַבֵּר,
כִּי שֹׁרֶשׁ שֶׁמְּקַבֵּל כָּבוֹד מַתְחִיל לְהִתְגַּבֵּר.

[בַּיִת 4 — פַּאנְצ׳ִים שֶׁל דּוֹרוֹת]
מִי שֶׁזּוֹכֵר אֶת הַשֵּׁם — מוֹצֵא מַפָּה בְּעָנָף,
מִי שֶׁמּוֹדֶה לַשֹּׁרֶשׁ — מְקַבֵּל רוּחַ בַּגַּב.
מִי שֶׁבּוֹנֶה עָתִיד בְּלִי עָבָר, בּוֹנֶה עַל חוֹל,
מִי שֶׁמְּכַבֵּד דּוֹרוֹת, מְקַבֵּל עֹמֶק בַּקּוֹל.
שֵׁם הוּא לֹא תָּג עַל דֶּלֶת, הוּא תֹּף עָמֹק בֶּחָזֶה,
הוּא מַזְכִּיר לִי: “בֵּן אָדָם, אַתָּה לֹא בָּאתָ מִזֶּה.”
אִם הָעֵץ מָלֵא שֵׁמוֹת, הַלֵּב מָלֵא שְׁלִיחוּת,
כָּל עָלֶה אוֹמֵר: תַּמְשִׁיךְ, אַךְ תִּשְׁמֹר עַל הַפְּשִׁיטוּת.
זֶה פַּאנְץ׳ שְׁבִיעִי: שֵׁם שֶׁל סָב לֹא נִשְׁאָר בְּמִסְגֶּרֶת,
הוּא צוֹעֵד בְּךָ בָּרְחוֹב וּמְיַשֵּׁר לְךָ אֶת הַדֶּרֶךְ.
זֶה פַּאנְץ׳ שְׁמִינִי: כָּל נֵר שֶׁאַתָּה מַדְלִיק לַדּוֹר,
מְחַזִּיר אוֹר לְמִי שֶׁהִדְלִיק אוֹתְךָ לִפְנֵי הָאוֹר.

[מַקְהֵלָה]
אוֹר, דּוֹר, שֹׁרֶשׁ, בַּיִת,
שֵׁם, נֵר, דָּם, בְּרִית.
אוֹר, דּוֹר, שֹׁרֶשׁ, בַּיִת,
עֵץ עִם שֵׁמוֹת — וְנֶפֶשׁ יְהוּדִית.

[דִּבּוּר לִפְנֵי סִיּוּם]
וְאִם שָׁאַלְתָּ מִי אַתָּה, אַל תַּעֲנֶה רַק בְּשֵׁם,
תַּעֲנֶה בְּקוֹל דּוֹרוֹת שֶׁעוֹבֵר דַּרְכְּךָ וְנוֹשֵׁם.
כִּי אָדָם בְּלִי זִכָּרוֹן הוּא כְּמוֹ שִׁיר בְּלִי פִּזְמוֹן,
וְאָדָם עִם שֹׁרֶשׁ חַי — מְהַלֵּךְ עִם אֲרוֹן אוֹר בַּלֵּב.
זֶה פַּאנְץ׳ תְּשִׁיעִי: הַשֵּׁם שֶׁקִּבַּלְתָּ הוּא לֹא סוֹף,
הוּא הַתְחָלָה שֶׁמְּבַקֶּשֶׁת מִמְּךָ לִצְמֹחַ וְלֶאֱסֹף.
זֶה פַּאנְץ׳ עֲשִׂירִי: כְּשֶׁהַלֵּב מְכַבֵּד אֶת הַקּוֹדֵם,
הַדּוֹר הַבָּא מְקַבֵּל גַּב, וְהַבַּיִת נוֹשֵׁם.

[פִּזְמוֹן אַחֲרוֹן]
כָּל שֵׁם בֶּעָנָף הוּא נֵר בְּתוֹךְ הַדָּם,
כָּל דּוֹר שֶׁהָיָה כָּאן עוֹד לוֹחֵשׁ לִי: קוּם גַּם.
אֲנִי לֹא רַק עַצְמִי, אֲנִי דֶּרֶךְ שֶׁעָבְרָה,
עֵץ עִם שֵׁמוֹת בַּלֵּב — וְשֹׁרֶשׁ שֶׁל בְּרָכָה.
כָּל שֵׁם בֶּעָנָף הוּא נֵר בְּתוֹךְ הַדָּם,
כָּל יֶלֶד שֶׁיִּוָּלֵד יִשְׁמַע: אַתָּה לֹא סְתָם.
אֲנִי לוֹקֵחַ אֶת הָאוֹר וּמַעֲבִיר אוֹתוֹ הָלְאָה,
בְּשֵׁם, בְּשִׁיר, בְּכָבוֹד — מִדּוֹר לְדוֹר עֲלִיָּה.

[Outro]
גִּיטָרָה נִרְגַּעַת, חָלִיל עוֹלֶה וְיוֹרֵד,
הָאַלְבּוֹם נִסְגָּר, אֲבָל הָעֵץ עוֹד עוֹמֵד.
אֲנִי כּוֹתֵב אֶת שְׁמִי בֶּעָנָף בְּלִי גַּאֲוָה,
וְכָל הָעֵץ מֵאִיר: שֵׁם, שֹׁרֶשׁ, דּוֹר, תּוֹדָה.
